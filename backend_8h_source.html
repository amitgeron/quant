133 static inline int w_sock_cmp(const struct w_sock * const a,
134  const struct w_sock * const b)
135 {
136  const uint32_t ap = ((uint32_t)a->hdr->udp.sport << 16) + a->hdr->udp.dport;
137  const uint32_t bp = ((uint32_t)b->hdr->udp.sport << 16) + b->hdr->udp.dport;
138  return (ap > bp) - (ap < bp);
139 }
