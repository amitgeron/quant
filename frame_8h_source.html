200 static inline bool __attribute__((nonnull))
201 is_ack_eliciting(const struct frames * const f)
202 {
203  static const struct frames ack_or_pad =
204  bitset_t_initializer(1 << FRM_ACK | 1 << FRM_PAD);
205  struct frames not_ack_or_pad = *f;
206  bit_nand(NUM_FRAM_TYPES, &not_ack_or_pad, &ack_or_pad);
207  return !bit_empty(NUM_FRAM_TYPES, &not_ack_or_pad);
208 }
