137 static inline epoch_t __attribute__((nonnull, always_inline))
138 strm_epoch(const struct q_stream * const s)
139 {
140  if (unlikely(s->id < 0))
141  switch (s->id) {
142  case -4:
143  return ep_init;
144  case -3:
145  return ep_0rtt;
146  case -2:
147  return ep_hshk;
148  case -1:
149  return ep_data;
150  default:
151  die("illegal sid %d", s->id);
152  }
153 
154  if (unlikely(s->c->state == conn_opng))
155  return ep_0rtt;
156 
157  return ep_data;
158 }
