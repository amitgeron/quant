213 static inline struct pn_space * __attribute__((always_inline, nonnull))
214 pn_for_epoch(struct q_conn * const c, const epoch_t e)
215 {
216  switch (e) {

218  return &c->pn_init.pn;

229 static inline epoch_t __attribute__((always_inline, nonnull))
230 epoch_in(const struct q_conn * const c)
231 {
232  const size_t epoch = ptls_get_read_epoch(c->tls.t);
233  switch (epoch) {
234  case 0:
235  return ep_init;
236  case 1:
237  return ep_0rtt;
238  case 2:
239  return ep_hshk;
240  case 3:
241  return ep_data;
242  default:
243  die("unhandled epoch %u", epoch);
244  }
245 }

258 cid_cmp(const struct cid * const a, const struct cid * const b)
259 {
