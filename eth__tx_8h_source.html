52 eth_tx(struct w_engine * const w, struct w_iov * const v, const uint16_t len)
53 {
54  // find a tx ring with space
55  struct netmap_ring * txr = 0;
56  for (uint32_t r = 0; likely(r < w->b->nif->ni_tx_rings); r++) {
57  txr = NETMAP_TXRING(w->b->nif, w->b->cur_txr);
58  if (likely(!nm_ring_empty(txr)))
59  // we have space in this ring
60  break;
61 
62  warn(INF, "tx ring %u full; moving to next", w->b->cur_txr);
63  w->b->cur_txr = (w->b->cur_txr + 1) % w->b->nif->ni_tx_rings;
64  txr = 0;
65  }
66 
67  // return false if all rings are full
68  if (unlikely(txr == 0)) {
69  warn(INF, "all tx rings are full");
70  return false;
71  }
72 
73  struct netmap_slot * const s = &txr->slot[txr->cur];
74  w->b->slot_buf[txr->ringid][txr->cur] = v;
75  s->len = len + sizeof(struct eth_hdr);
76 
77  if (unlikely(nm_ring_space(txr) == 1 || sq_next(v, next) == 0)) {
78  // we are using the last slot in this ring, or this is the last w_iov in
79  // this batch - mark the slot for reporting
80  s->flags = NS_REPORT | NS_BUF_CHANGED;
81  } else
82  s->flags = NS_BUF_CHANGED;
83 
84  warn(DBG, "placing iov idx %u into tx ring %u slot %d (swap with %u)",
85  v->idx, w->b->cur_txr, txr->cur, s->buf_idx);
86  // temporarily place v into the current tx ring
87  const uint32_t slot_idx = s->buf_idx;
88  s->buf_idx = v->idx;
89  v->idx = slot_idx;
90

95  warn(DBG, "Eth %s -> %s, type %d, len %lu", ether_ntoa_r(&eth->src, src),
