197 static int encrypt_ticket_cb(ptls_encrypt_ticket_t *_self, ptls_t *tls, int is_encrypt, ptls_buffer_t *dst, ptls_iovec_t src)
198 {
199  struct st_util_session_cache_t *self = (void *)_self;
200  int ret;
201 
202  if (is_encrypt) {
203 
204  /* replace the cached entry along with a newly generated session id */
205  free(self->data.base);
206  if ((self->data.base = malloc(src.len)) == NULL)
207  return PTLS_ERROR_NO_MEMORY;
208 
209  ptls_get_context(tls)->random_bytes(self->id, sizeof(self->id));
210  memcpy(self->data.base, src.base, src.len);
211  self->data.len = src.len;
212 
213  /* store the session id in buffer */
214  if ((ret = ptls_buffer_reserve(dst, sizeof(self->id))) != 0)
215  return ret;
216  memcpy(dst->base + dst->off, self->id, sizeof(self->id));
217  dst->off += sizeof(self->id);
218 
219  } else {
220 
221  /* check if session id is the one stored in cache */
222  if (src.len != sizeof(self->id))
223  return PTLS_ERROR_SESSION_NOT_FOUND;
224  if (memcmp(self->id, src.base, sizeof(self->id)) != 0)
225  return PTLS_ERROR_SESSION_NOT_FOUND;
226 
227  /* return the cached value */
228  if ((ret = ptls_buffer_reserve(dst, self->data.len)) != 0)
229  return ret;
230  memcpy(dst->base + dst->off, self->data.base, self->data.len);
231  dst->off += self->data.len;
232  }
233 
234  return 0;
235 }

248 static inline int resolve_address(struct sockaddr *sa, socklen_t *salen, const char *host, const char *port, int family, int type,
249  int proto)
250 {
251  struct addrinfo hints, *res;
252  int err;
253 
254  memset(&hints, 0, sizeof(hints));
255  hints.ai_family = family;
256  hints.ai_socktype = type;
257  hints.ai_protocol = proto;
258  hints.ai_flags = AI_ADDRCONFIG | AI_NUMERICSERV | AI_PASSIVE;
259  if ((err = getaddrinfo(host, port, &hints, &res)) != 0 || res == NULL) {
260  fprintf(stderr, "failed to resolve address:%s:%s:%s\n", host, port,
261  err != 0 ? gai_strerror(err) : "getaddrinfo returned NULL");
262  return -1;
263  }
264 
265  memcpy(sa, res->ai_addr, res->ai_addrlen);
266  *salen = res->ai_addrlen;
267 
268  freeaddrinfo(res);
269  return 0;
270 }

272 static inline int normalize_txt(uint8_t *p, size_t len)
273 {
274  uint8_t *const end = p + len, *dst = p;
275 
276  if (p == end)
277  return 0;
278 
279  do {
280  size_t block_len = *p++;
281  if (end - p < block_len)
282  return 0;
283  memmove(dst, p, block_len);
284  dst += block_len;
285  p += block_len;
286  } while (p != end);
287  *dst = '\0';
288 
289  return 1;
290 }

292 static inline ptls_iovec_t resolve_esni_keys(const char *server_name)
293 {
294  char esni_name[256], *base64;
295  uint8_t answer[1024];
296  ns_msg msg;
297  ns_rr rr;
298  ptls_buffer_t decode_buf;
299  ptls_base64_decode_state_t ds;
300  int answer_len;
301 
302  ptls_buffer_init(&decode_buf, "", 0);
303 
304  if (snprintf(esni_name, sizeof(esni_name), "_esni.%s", server_name) > sizeof(esni_name) - 1)
305  goto Error;
306  if ((answer_len = res_query(esni_name, ns_c_in, ns_t_txt, answer, sizeof(answer))) <= 0)
307  goto Error;
308  if (ns_initparse(answer, answer_len, &msg) != 0)
309  goto Error;
310  if (ns_msg_count(msg, ns_s_an) < 1)
311  goto Error;
312  if (ns_parserr(&msg, ns_s_an, 0, &rr) != 0)
313  goto Error;
314  base64 = (void *)ns_rr_rdata(rr);
315  if (!normalize_txt((void *)base64, ns_rr_rdlen(rr)))
316  goto Error;
317 
318  ptls_base64_decode_init(&ds);
319  if (ptls_base64_decode(base64, &ds, &decode_buf) != 0)
320  goto Error;
321  assert(decode_buf.is_allocated);
322 
323  return ptls_iovec_init(decode_buf.base, decode_buf.off);
324 Error:
325  ptls_buffer_dispose(&decode_buf);
326  return ptls_iovec_init(NULL, 0);
327 }
