62 // Computes number of bytes need to enccode @p v in QUIC varint encoding.
68 static inline uint8_t __attribute__((const)) varint_size(const uint64_t v)
69 {
70  if (v <= VARINT1_MAX)
71  return 1;
72  if (v <= VARINT2_MAX)
73  return 2;
74  if (v <= VARINT4_MAX)
75  return 4;
76  return 8;
77 }

129 static inline void __attribute__((nonnull))
130 encv(uint8_t ** pos, const uint8_t * const end, const uint64_t val)
131 {
132  ensure((val & VARINT_MASK) == 0, "value overflow: %" PRIu64, val);
133 
134  if ((val & VARINT_MASK8) != 0) {
135  ensure(*pos + 8 <= end, "buffer overflow: %lu", end - *pos);
136  *(*pos + 0) = ((val >> 56) & 0x3f) + 0xc0;
137  *(*pos + 1) = (val >> 48) & 0xff;
138  *(*pos + 2) = (val >> 40) & 0xff;
139  *(*pos + 3) = (val >> 32) & 0xff;
140  *(*pos + 4) = (val >> 24) & 0xff;
141  *(*pos + 5) = (val >> 16) & 0xff;
142  *(*pos + 6) = (val >> 8) & 0xff;
143  *(*pos + 7) = val & 0xff;
144  *pos += 8;
145  return;
146  }
147 
148  if ((val & VARINT_MASK4) != 0) {
149  ensure(*pos + 4 <= end, "buffer overflow: %lu", end - *pos);
150  *(*pos + 0) = ((val >> 24) & 0x3f) + 0x80;
151  *(*pos + 1) = (val >> 16) & 0xff;
152  *(*pos + 2) = (val >> 8) & 0xff;
153  *(*pos + 3) = val & 0xff;
154  *pos += 4;
155  return;
156  }
157 
158  if ((val & VARINT_MASK2) != 0) {
159  ensure(*pos + 2 <= end, "buffer overflow: %lu", end - *pos);
160  *(*pos + 0) = ((val >> 8) & 0x3f) + 0x40;
161  *(*pos + 1) = val & 0xff;
162  *pos += 2;
163  return;
164  }
165 
166  ensure(*pos + 1 <= end, "buffer overflow: %lu", end - *pos);
167  **pos = val & 0x3f;
168  *pos += 1;
169 }

172 static inline void __attribute__((nonnull)) encvl(uint8_t ** pos,
173  const uint8_t * const end,
174  const uint64_t val,
175  const uint8_t len)
176 {
177  const uint8_t len_needed = varint_size(val);
178  ensure(len_needed <= len, "value/len mismatch");
179 
180  if (len_needed == len) {
181  encv(pos, end, val);
182  return;
183  }
184 
185  if (len == 2) {
186  enc1(pos, end, 0x40);
187  enc1(pos, end, (uint8_t)val);
188  return;
189  }
190 
191  if (len == 4) {
192  enc1(pos, end, 0x80);
193  enc1(pos, end, 0x00);
194  enc2(pos, end, (uint16_t)val);
195  return;
196  }
197 
198  if (len == 8) {
199  enc1(pos, end, 0xC0);
200  enc1(pos, end, 0x00);
201  enc2(pos, end, 0x00);
202  enc4(pos, end, (uint32_t)val);
203  return;
204  }
205 }

282 static inline bool __attribute__((nonnull)) decv(uint64_t * const val,
283  const uint8_t ** const pos,
284  const uint8_t * const end)
285 {
286  switch (**pos & 0xc0) {
287  case 0xc0:
288  if (unlikely(*pos + 8 > end))
289  return false;
290  *val =
291  ((uint64_t)(*(*pos + 0) & 0x3f) << 56) +
292  ((uint64_t)(*(*pos + 1)) << 48) + ((uint64_t)(*(*pos + 1)) << 48) +
293  ((uint64_t)(*(*pos + 2)) << 40) + ((uint64_t)(*(*pos + 3)) << 32) +
294  ((uint64_t)(*(*pos + 4)) << 24) + ((uint64_t)(*(*pos + 5)) << 16) +
295  ((uint64_t)(*(*pos + 6)) << 8) + ((uint64_t)(*(*pos + 7)) << 0);
296  *pos += 8;
297  return true;
298 
299  case 0x80:
300  if (unlikely(*pos + 4 > end))
301  return false;
302  *val = ((uint64_t)(*(*pos + 0) & 0x3f) << 24) +
303  ((uint64_t)(*(*pos + 1)) << 16) +
304  ((uint64_t)(*(*pos + 2)) << 8) + ((uint64_t)(*(*pos + 3)) << 0);
305  *pos += 4;
306  return true;
307 
308  case 0x40:
309  if (unlikely(*pos + 2 > end))
310  return false;
311  *val = ((uint64_t)(*(*pos + 0) & 0x3f) << 8) + (uint64_t)(*(*pos + 1));
312  *pos += 2;
313  return true;
314 
315  case 0x00:
316  if (unlikely(*pos + 1 > end))
317  return false;
318  *val = (**pos) & 0x3f;
319  *pos += 1;
320  return true;
321  }
322 
323  return false;
324 }
