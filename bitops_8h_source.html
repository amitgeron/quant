212  volatile const uint8_t *tab, uint32_t n)
213 {

219  uint32_t i;
220 
221  for (i = 0; i < n; i++)
222  {
223  uint8_t item = tab[i];

225  mask = mask_u8(*a, i); ra = (ra & ~mask) | (item & mask);
226  mask = mask_u8(*b, i); rb = (rb & ~mask) | (item & mask);
227  mask = mask_u8(*c, i); rc = (rc & ~mask) | (item & mask);

229  }

284 static inline void copy_bytes_unaligned(uint8_t *out, const uint8_t *in, size_t len, uint8_t offset)
285 {
286  uint8_t byte_off = offset / 8;
287  uint8_t bit_off = offset & 7;
288  uint8_t rmask = (1 << bit_off) - 1;
289  uint8_t lmask = ~rmask;
290  size_t i;
291 
292  for (i = 0; i < len; i++)
293  {
294  out[i] = (in[i + byte_off] << bit_off) & lmask;
295  out[i] |= (in[i + byte_off + 1] >> (8 - bit_off)) & rmask;
296  }
297 }
