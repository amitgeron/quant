168 #define bps(bytes, secs) \
169  __extension__({ \
170  static char _str[32]; \
171  const double _bps = \
172  (bytes) && (fpclassify(secs) != FP_ZERO) ? (bytes)*8 / (secs) : 0; \
173  if (_bps > 1000000000) \
174  snprintf(_str, sizeof(_str), "%.3f Gb/s", _bps / 1000000000); \
175  else if (_bps > 1000000) \
176  snprintf(_str, sizeof(_str), "%.3f Mb/s", _bps / 1000000); \
177  else if (_bps > 1000) \
178  snprintf(_str, sizeof(_str), "%.3f Kb/s", _bps / 1000); \
179  else \
180  snprintf(_str, sizeof(_str), "%.3f b/s", _bps); \
181  _str; \
182  })
