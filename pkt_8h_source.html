126 static inline const char * __attribute__((const, nonnull))
127 pkt_type_str(const uint8_t flags, const void * const vers)
128 {
129  if (is_lh(flags)) {
130  if (((const uint8_t * const)vers)[0] == 0 &&
131  ((const uint8_t * const)vers)[1] == 0 &&
132  ((const uint8_t * const)vers)[2] == 0 &&
133  ((const uint8_t * const)vers)[3] == 0)
134  return "Version Negotiation";
135  switch (pkt_type(flags)) {
136  case LH_INIT:
137  return "Initial";
138  case LH_RTRY:
139  return "Retry";
140  case LH_HSHK:
141  return "Handshake";
142  case LH_0RTT:
143  return "0-RTT Protected";
144  }
145  } else if (pkt_type(flags) == SH)
146  return "Short";
147  return RED "Unknown" NRM;
148 }
